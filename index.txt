    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const params = new URLSearchParams(window.location.search);
            const sender = params.get('sender');

            if (sender) {
                document.getElementById('inputSection').style.display = 'none';
                document.getElementById('wishesSection').style.display = 'block';
                document.getElementById('senderNameDisplay').innerHTML = `By ${decodeURIComponent(sender)}`;
            } else {
                document.getElementById('inputSection').style.display = 'block';
                document.getElementById('wishesSection').style.display = 'none';
            }
        });

        function generateLinkAndDisplayOptions(inputElementId) {
            const userNameInput = document.getElementById(inputElementId);
            const userName = userNameInput.value.trim();

            if (!userName) {
                alert('Kripya apna naam enter karein!');
                return;
            }

            const encodedName = encodeURIComponent(userName);
            const baseUrl = window.location.origin + window.location.pathname;
            const shareLink = `${baseUrl}?sender=${encodedName}`;
            
            let linkDisplayId, outputContainerId;
            if (inputElementId === 'userName') {
                linkDisplayId = 'generatedLinkInitial';
                outputContainerId = 'outputContainerInitial';
                document.getElementById('initialPrompt').style.display = 'none';
                const yourNameDisplay = document.getElementById('yourNameDisplay');
                yourNameDisplay.innerHTML = `Happy Navaratri By ${userName}`;
                yourNameDisplay.style.display = 'block';
            } else {
                linkDisplayId = 'generatedLinkReply';
                outputContainerId = 'outputContainerReply';
            }

            const linkDisplay = document.getElementById(linkDisplayId);
            linkDisplay.innerText = shareLink;
            document.getElementById(outputContainerId).style.display = 'block';
        }

        function shareOnWhatsApp(inputElementId) {
            const userNameInput = document.getElementById(inputElementId);
            const userName = userNameInput.value.trim();
            if (!userName) {
                alert('Pehle link generate karein!');
                return;
            }
            const encodedName = encodeURIComponent(userName);
            const baseUrl = window.location.origin + window.location.pathname;
            const shareLink = `${baseUrl}?sender=${encodedName}`;
            const message = `*Happy Navaratri By ${userName}!* âœ¨\n\nIs link par click karein aur shubh kamnayein prapt karein:\n${shareLink}`;
            const whatsappUrl = `https://api.whatsapp.com/send?text=${encodeURIComponent(message)}`;
            window.open(whatsappUrl, '_blank');
        }
        
        // **UPDATED COPY FUNCTION**
        function copyLink(linkSpanId) {
            const linkText = document.getElementById(linkSpanId).innerText;
            
            // Modern way: using navigator.clipboard (works on HTTPS sites)
            if (navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(linkText).then(() => {
                    alert('Copied to clipboard');
                }).catch(err => {
                    alert('Link copy nahi ho paya.');
                });
            } else {
                // Fallback way: for local files (file://) or HTTP sites
                const textArea = document.createElement('textarea');
                textArea.value = linkText;
                textArea.style.position = 'absolute';
                textArea.style.left = '-9999px'; // Hide it
                document.body.appendChild(textArea);
                textArea.select();
                try {
                    document.execCommand('copy');
                    alert('Copied to clipboard');
                } catch (err) {
                    alert('Is browser mein automatic copy support nahi hai.');
                } finally {
                    document.body.removeChild(textArea);
                }
            }
        }
